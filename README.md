public class AbsoluteValue
{
    public void check()
    {
        int i,k;
        i=5;
        k=i<0?-i:i;  //To get the absolute value
        System.out.println("The absolute value of " + i  + " is " + k);
        i=-5;
        k=i<0?-i:i;  //To get the absolute value
        System.out.println("The Absolute value of " + i + " is " + k);
    }
}
